# Colors

Our color system is built with CSS custom properties, making it easy to maintain consistency
across the design system.

export const ColorSwatch = ({ name, variable, value }) => (
  <div
    style={{
      display: 'flex',
      alignItems: 'center',
      gap: '1rem',
      padding: '0.75rem',
      borderRadius: '0.5rem',
      backgroundColor: 'var(--color-surface)',
      border: '1px solid var(--color-border-subtle)',
      marginBottom: '0.5rem',
    }}
  >
    <div
      style={{
        width: '3rem',
        height: '3rem',
        borderRadius: '0.5rem',
        backgroundColor: value,
        border: '1px solid var(--color-border)',
        flexShrink: 0,
      }}
    />
    <div style={{ flex: 1 }}>
      <div
        style={{
          fontWeight: 600,
          fontSize: '0.875rem',
          color: 'var(--color-text-primary)',
        }}
      >
        {name}
      </div>
      <code
        style={{
          fontSize: '0.75rem',
          color: 'var(--color-text-secondary)',
          fontFamily: 'var(--font-family-mono)',
        }}
      >
        {variable}
      </code>
    </div>
    <code
      style={{
        fontSize: '0.75rem',
        color: 'var(--color-text-muted)',
        fontFamily: 'var(--font-family-mono)',
      }}
    >
      {value}
    </code>
  </div>
)

export const ColorPalette = ({ title, colors }) => (
  <div style={{ marginBottom: '2rem' }}>
    <h3
      style={{
        marginBottom: '1rem',
        color: 'var(--color-text-primary)',
        fontSize: '1.125rem',
        fontWeight: 600,
      }}
    >
      {title}
    </h3>
    <div>
      {colors.map((color) => (
        <ColorSwatch key={color.name} {...color} />
      ))}
    </div>
  </div>
)

## Usage

```css
/* Use semantic colors for consistent styling */
.card {
  background-color: var(--color-surface);
  border: 1px solid var(--color-border);
  color: var(--color-text-primary);
}

/* Use palette colors for specific brand elements */
.brand-accent {
  background-color: var(--color-primary-500);
}

/* Use status colors for feedback states */
.alert-error {
  background-color: var(--color-error-subtle);
  border: 1px solid var(--color-error);
  color: var(--color-error);
}

.alert-success {
  background-color: var(--color-success-subtle);
  border: 1px solid var(--color-success);
  color: var(--color-success);
}
```
